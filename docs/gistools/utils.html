 28def has_method(arg, method):
 29	"""Checks if an object has a callable method with the given name.
 30
 31	Args:
 32		arg: The object to check for the method.
 33		method: The name of the method to look for.
 34
 35	Returns:
 36		True if the object has a method with the given name that is also callable (a function), False otherwise.
 37	"""	
 38	# This function uses two checks:
 39	# 1. hasattr(arg, method): Checks if the object has an attribute with the given method name.
 40	# 2. callable(getattr(arg, method)): Checks if the retrieved attribute is actually callable, meaning it's a function. 
 41	return hasattr(arg, method) and callable(getattr(arg, method))

 43def get_class_name(obj):
 44	"""Retrieves the class name of the given object.
 45
 46	This function takes an object of any type as input and returns the name
 47	of the class to which it belongs. This can be useful for introspection
 48	or debugging purposes, where you might need to determine the type
 49	of an object at runtime.
 50
 51	Args:
 52		obj (object): The object whose class name you want to obtain.
 53
 54	Returns:
 55		str: The name of the class to which the object belongs.
 56	"""	
 57	return type(obj).__name__

 59def get_class_attr(classname):
 60	"""Retrieves public attributes of a given class.
 61
 62	This function takes the name of a class (`classname`) as input and returns
 63	a list of its public attributes (excluding methods and dunder methods).
 64	It uses `inspect.getmembers` to introspect the class and filters out
 65	methods using `inspect.isroutine` and dunder methods (those starting and
 66	ending with double underscores) using list comprehension.
 67
 68	Args:
 69		classname (str): The name of the class to introspect.
 70
 71	Returns:
 72		list: A list of public attributes (strings) of the class.
 73	"""	
 74	attributes = inspect.getmembers(classname, lambda x : not(inspect.isroutine(x)))
 75	return [x for x in attributes if not(x[0].startswith('__') and x[0].endswith('__'))]

238def timestr2seconds(timestr):
239	"""
240	Converts a time string in HH:MM:SS format to seconds.
241
242	Args:
243		timestr: The time string in HH:MM:SS format.
244
245	Returns:
246		The time in seconds.
247
248	Example:
249		timestr2seconds("01:02:03")
250		# Output: 3723
251	"""
252	ftr = [3600,60,1]
253	s = sum([a*b for a, b in zip(ftr, [int(i) for i in timestr.split(":")])])

479def is_numeric(literal):
480	"""
481	Checks if a given string represents a numeric value.
482	Credits: http://www.rosettacode.org/wiki/Determine_if_a_string_is_numeric#Python
483
484	Args:
485		literal: The string to check.
486
487	Returns:
488		True if the string represents a numeric value, False otherwise.
489
490	Example:
491		is_numeric("123")     # True (integer)
492		is_numeric("12.34")   # True (float)
493		is_numeric("1.23e+5") # True (scientific notation)
494		is_numeric("0x1A")    # True (hexadecimal)
495		is_numeric("0b101")   # True (binary)
496		is_numeric("0o123")   # True (octal)
497		is_numeric("abc")     # False
498	"""
499	castings = [int, float, complex,
500			lambda s: int(s, 2), #binary
501			lambda s: int(s, 8), #octal
502			lambda s: int(s,16)  #hex
503		]
504	for cast in castings:
505		try:
506			cast(literal)
507			return True
508		except ValueError:
509			pass
510
511	return False
