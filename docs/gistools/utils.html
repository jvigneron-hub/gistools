  28def has_method(arg, method):
  29	"""Checks if an object has a callable method with the given name.
  30
  31	Args:
  32		arg: The object to check for the method.
  33		method: The name of the method to look for.
  34
  35	Returns:
  36		True if the object has a method with the given name that is also callable (a function), False otherwise.
  37	"""	
  38	# This function uses two checks:
  39	# 1. hasattr(arg, method): Checks if the object has an attribute with the given method name.
  40	# 2. callable(getattr(arg, method)): Checks if the retrieved attribute is actually callable, meaning it's a function. 
  41	return hasattr(arg, method) and callable(getattr(arg, method))

  59def get_class_attr(classname):
  60	"""Retrieves public attributes of a given class.
  61
  62	This function takes the name of a class (`classname`) as input and returns
  63	a list of its public attributes (excluding methods and dunder methods).
  64	It uses `inspect.getmembers` to introspect the class and filters out
  65	methods using `inspect.isroutine` and dunder methods (those starting and
  66	ending with double underscores) using list comprehension.
  67
  68	Args:
  69		classname (str): The name of the class to introspect.
  70
  71	Returns:
  72		list: A list of public attributes (strings) of the class.
  73	"""	
  74	attributes = inspect.getmembers(classname, lambda x : not(inspect.isroutine(x)))
  75	return [x for x in attributes if not(x[0].startswith('__') and x[0].endswith('__'))]

 508		except ValueError:
 509			pass
