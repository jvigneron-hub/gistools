 133def has_method(arg, method):
 134	"""Checks if an object has a callable method with the given name.
 135
 136	Args:
 137	- **arg**: The object to check for the method.
 138	- **method**: The name of the method to look for.

 141	- **boolean**: True if the object has a method with the given name that is also callable (a function), False otherwise.

 165def get_class_attr(classname):
 166	"""Retrieves public attributes of a given class.
 167
 168	This function takes the name of a class (`classname`) as input and returns
 169	a list of its public attributes (excluding methods and dunder methods).
 170	It uses `inspect.getmembers` to introspect the class and filters out
 171	methods using `inspect.isroutine` and dunder methods (those starting and
 172	ending with double underscores) using list comprehension.
 173
 174	Args:
 175	- **classname (str)**: The name of the class to introspect.
 176
 177	Returns:
 178	- **list**: A list of public attributes (strings) of the class.
 179	"""	
 180	attributes = inspect.getmembers(classname, lambda x : not(inspect.isroutine(x)))
 181	return [x for x in attributes if not(x[0].startswith('__') and x[0].endswith('__'))]
